function calculateFinalAmount(monto, fechaPago) {
    let paymentDay = new Date(fechaPago).getDate();
    if (paymentDay >= 1 && paymentDay <= 5) {
        return monto * 0.95;  // 5% discount
    } else if (paymentDay >= 6 && paymentDay <= 15) {
        return monto;  // No change
    } else {
        return monto * 1.05;  // 5% increase
    }
}
